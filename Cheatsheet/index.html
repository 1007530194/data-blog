NumPy

array initialization

import numpy as np

a = np.array([2, 3, 4])                # direct initialization
print(a)
print(np.empty(20, dtype=np.float32))  # single precision array with 20 entries
print(np.zeros(200))                   # initialize 200 zeros
print(np.ones((3,3), dtype=np.int32))  # 3 x 3 integer matrix with ones
print(np.eye(200))                     # ones on the diagonal
print(np.zeros_like(a))                # returns array with zeros and the shape of a
print(np.linspace(0., 10., 100))       # 100 points from 0 to 10
print(np.arange(0, 100, 2))            # points from 0 to <100 with step width 2
print(np.logspace(-5, 2, 100))         # 100 log-spaced points between 1e-5 and 1e2
print(np.copy(a))                      # copy array to new memory

array properties and operations

a = np.array([[3,2,1],[6,5,4]])
print(a)
print(a.shape)                # a tuple with the lengths of each axis
print(len(a))                 # length of axis 0
print(a.ndim)                 # number of dimensions (axes)
a.sort(axis=1)                # sort array along axis
print(a)
print(a.flatten())            # collapse array to one dimension
print(a.conj())               # return complex conjugate
print(a.astype(np.int16))     # cast to integer
print(np.argmax(a, axis=1))   # return index of maximum along a given axis
print(np.cumsum(a))           # return cumulative sum
print(np.any(a))              # True if any element is True
print(np.all(a))              # True if all elements are True
print(np.argsort(a, axis=1))  # return sorted index arr
